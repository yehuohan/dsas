133 template <typename T>
134 void BTree<T>::solve_overflow(BTNodePtr<T> node)
135 {
136  // 根据分支数判断是否上溢
137  if (node->child.size() <= this->m_order)
138  return;
139 
140  // 判断是否为根节点
141  BTNodePtr<T> p = node->parent;
142  if (!p)
143  {
144  // 在在根节点上溢时，树高度会增加
145  this->m_root = p = new BTNode<T>(this->m_order); // BTNode默认构造函数，在没关键码时有一个指向nullptr子节点
146  p->child[0] = node;
147  node->parent = p;
148  }
149 
150  // 转移node中s之后关键码和分支节点，追加到新节点rc中
151  // 即转移示意图中的 *[5]* 部分
152  int s = this->m_order / 2; // 确定上移的节点位置
153  BTNodePtr<T> rc = new BTNode<T>(this->m_order); // BTNode默认构造函数，在没关键码时有一个指向nullptr子节点
154  rc->child.clear(); // 清除第一个默认添加的nullptr子节点
155  for (int k = s + 1; k < node->key.size(); k ++)
156  {
157  rc->key.push_back(node->key[k]);
158  rc->child.push_back(node->child[k]);
159  }
160  rc->child.push_back(node->child[node->child.size()-1]); // 分支节点数比关键码数多1
161  if (rc->child[0]) // 非叶节点的子节点均在存(不为nullptr)
162  {
163  for (int k = 0; k < rc->child.size(); k ++)
164  rc->child[k]->parent = rc; // 更新父节点，rc即在p->child中
165  }
166  node->key.remove(s+1, node->key.size());// 移除原节点node中的关键码和分支
167  node->child.remove(s+1, node->child.size());
168 
169  // 上移目标关键码
170  // 即上移示意图的中 [3]
171  int r = p->key.search(node->key[s]); // 确定移至父节点的位置
172  p->key.insert(r+1, node->key.remove(s));
173  p->child.insert(r+2, rc); // 连接节点，p->child[r+2]现在指向rc
174  rc->parent = p;
175 
176  solve_overflow(p);
177 }

216 template <typename T>
217 void BTree<T>::solve_underflow(BTNodePtr<T> node)
218 {
219  int order_floor = (this->m_order+1)/2;
220  if (node->child.size() >= order_floor)
221  return;
222 
223  BTNodePtr<T> p = node->parent;
224  if (!p)
225  {
226  if ((0 == node->key.size()) && node->child[0]) // 根节点不含关键码(即没有数据了)
227  {
228  // 在在根节点下溢时，树高度会下降
229  this->m_root = node->child[0];
230  this->m_root->parent = nullptr;
231  node->child[0] = nullptr;
232  delete node;
233  }
234  return;
235  }
236 
237  // 获取node在父节点中的位置，这里不能用search，因为node是指针，在vector中是无序的
238  // r在p->child的左右两侧，必有一侧存在
239  int r = p->child.find(node);
240 
241  // 尝试旋转
242  if (r > 0)
243  {
244  // 尝试旋转，向左侧子节点借
245  BTNodePtr<T> s = p->child[r-1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
246  if (s->child.size() > order_floor)
247  {
248  //int y = r-1;
249  node->key.push_front(p->key[r-1]);
250  node->child.push_front(s->child.remove(s->child.size()-1));
251  p->key[r-1] = s->key.remove(s->key.size()-1);
252  return;
253  }
254  }
255  if (r + 1 < p->child.size())
256  {
257  // 尝试旋转，向右侧子节点借
258  BTNodePtr<T> s = p->child[r+1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
259  if(s->child.size() > order_floor)
260  {
261  //int y = r;
262  node->key.push_back(p->key[r]);
263  node->child.push_back(s->child.remove(0));
264  p->key[r] = s->key.remove(0);
265  return;
266  }
267  }
268 
269  // 使用合并
270  if (r > 0)
271  {
272  // 左侧存在
273  BTNodePtr<T> s = p->child[r-1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
274  //int y = r-1;
275 
276  s->key.push_back(p->key.remove(r-1));
277  p->child.remove(r);
278 
279  for (int k = 0; k < node->key.size(); k++)
280  s->key.push_back(node->key[0]);
281 
282  for (int k = 0; k < node->child.size(); k++)
283  {
284  if (node->child[k])
285  node->child[k]->parent = s;
286  s->child.push_back(node->child[k]);
287  }
288  delete node;
289  }
290  else
291  {
292  // 右侧存在
293  BTNodePtr<T> s = p->child[r+1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
294  //int y = r;
295 
296  s->key.push_front(p->key.remove(r));
297  p->child.remove(r);
298 
299  for (int k = 0; k < node->key.size(); k++)
300  s->key.insert(k, node->key[k]);
301 
302  for (int k = 0; k < node->child.size(); k++)
303  {
304  if (node->child[k])
305  node->child[k]->parent = s;
306  s->child.insert(k, node->child[k]);
307  }
308  delete node;
309  }
310 
311  solve_underflow(p);
312 }

371 {

373  if (node) return false;
374

393 template <typename T>
394 bool BTree<T>::remove(const T& e)
395 {
396  BTNodePtr<T> node = this->search(e);
397  if (!node) return false;
398 
399  int r = node->key.search(e);
400  if (node->child[0])
401  {
402  // node不是叶节点，则e右侧必有子节点
403  BTNodePtr<T> u = node->child[r+1];
404  while(node->child[0]) u = u->child[0]; // 迭代至e的后继(后继必定是叶节点)
405  node->key[r] = u->key[0]; // 将u->key[0]与待删除e交换位置
406  //u->key[0] = e; // 赋不赋值无所谓，反正要删除
407  node = u; // node重新指向包含待删除目标e的叶节点u
408  r = 0; // r也指向e在node->key中的下标
409  }
410 
411  // 如果待删除目标e在叶节点中，直接删除
412  node->key.remove(r);
413  node->child.remove(node->child.size() - 1); // 因为叶节点所有子节点为nullptr，故随便删除一个即可
414  this->m_size--;
415 
416  this->solve_underflow(node);
417  return true;
418 }
