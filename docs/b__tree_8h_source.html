105 template <typename T>
106 void BTree<T>::solve_overflow(BTNode<T>* node)
107 {
108  // 根据分支数判断是否上溢
109  if (node->child.size() <= this->m_order)
110  return;
111 
112  // 判断是否为根节点
113  BTNode<T>* p = node->parent;
114  if (!p)
115  {
116  // 在在根节点上溢时，树高度会增加
117  this->m_root = p = new BTNode<T>(); // BTNode默认构造函数，在没关键码时有一个指向nullptr子节点
118  p->child[0] = node;
119  node->parent = p;
120  }
121 
122  // 转移node中s之后关键码和分支节点，追加到新节点rc中
123  // 即转移示意图中的 *[5]* 部分
124  int s = this->m_order / 2; // 确定上移的节点位置
125  BTNode<T>* rc = new BTNode<T>(); // BTNode默认构造函数，在没关键码时有一个指向nullptr子节点
126  rc->child.clear(); // 清除第一个默认添加的nullptr子节点
127  for (int k = s + 1; k < node->key.size(); k ++)
128  {
129  rc->key.push_back(node->key[k]);
130  rc->child.push_back(node->child[k]);
131  }
132  rc->child.push_back(node->child[node->child.size()-1]); // 分支节点数比关键码数多1
133  if (rc->child[0]) // 非叶节点的子节点均在存(不为nullptr)
134  {
135  for (int k = 0; k < rc->child.size(); k ++)
136  rc->child[k]->parent = rc; // 更新父节点，rc即在p->child中
137  }
138  node->key.remove(s+1, node->key.size());// 移除原节点node中的关键码和分支
139  node->child.remove(s+1, node->child.size());
140 
141  // 上移目标关键码
142  // 即上移示意图的中 [3]
143  int r = p->key.search(node->key[s]); // 确定移至父节点的位置
144  p->key.insert(r+1, node->key.remove(s));
145  p->child.insert(r+2, rc); // 连接节点，p->child[r+2]现在指向rc
146  rc->parent = p;
147 
148  solve_overflow(p);
149 }

189 void BTree<T>::solve_underflow(BTNode<T>* node)
190 {
191  int order_floor = (this->m_order+1)/2;
192  if (node->child.size() >= order_floor)
193  return;
194 
195  BTNode<T>* p = node->parent;
196  if (!p)
197  {
198  if ((0 == node->key.size()) && node->child[0]) // 根节点不含关键码(即没有数据了)
199  {
200  // 在在根节点下溢时，树高度会下降
201  this->m_root = node->child[0];
202  this->m_root->parent = nullptr;
203  node->child[0] = nullptr;
204  delete node;
205  }
206  return;
207  }
208 
209  // 获取node在父节点中的位置，这里不能用search，因为node是指针，在vector中是无序的
210  // r在p->child的左右两侧，必有一侧存在
211  int r = p->child.find(node);
212 
213  // 尝试旋转
214  if (r > 0)
215  {
216  // 尝试旋转，向左侧子节点借
217  BTNode<T>* s = p->child[r-1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
218  if (s->child.size() > order_floor)
219  {
220  //int y = r-1;
221  node->key.push_front(p->key[r-1]);
222  node->child.push_front(s->child.remove(s->child.size()-1));
223  p->key[r-1] = s->key.remove(s->key.size()-1);
224  return;
225  }
226  }
227  if (r + 1 < p->child.size())
228  {
229  // 尝试旋转，向右侧子节点借

232  {
233  //int y = r;
234  node->key.push_back(p->key[r]);
235  node->child.push_back(s->child.remove(0));
236  p->key[r] = s->key.remove(0);
237  return;
238  }
239  }
240 
241  // 使用合并
242  if (r > 0)
243  {
244  // 左侧存在
245  BTNode<T>* s = p->child[r-1]; // p->child和node均为p的节点，node不为nullptr，其它的child也必定不为nullptr
246  //int y = r-1;
247 
248  s->key.push_back(p->key.remove(r-1));
249  p->child.remove(r);
250 
251  for (int k = 0; k < node->key.size(); k++)
252  s->key.push_back(node->key[0]);
253  //while(!node->key.is_empty())
254  // s->key.push_back(node->key.remove(0));
255 
256  for (int k = 0; k < node->child.size(); k++)
257  {
258  if (node->child[k])
259  node->child[k]->parent = s;
260  s->child.push_back(node->child[k]);
261  }
262  //while(!node->child.is_empty())
263  //{
264  // if (node->child[0])
265  // node->child[0]->parent = s;
266  // s->child.push_back(node->child.remove(0));
267  //}
268

271  node->child.clear(); //不用while，需要clear
272  delete node;
273  }

329 BTNode<T>* BTree<T>::search(const T& e)
330 {
331  BTNode<T>* node = this->m_root;
332  this->m_hot = nullptr;
333  while(node)
334  {
335  int r = node->key.search(e); // 根据有序性查找，返回不大于查找目标的下标
336  // 在insert使用BTree::search时，即可按序插入
337  if (0 <= r && e == node->key[r]) return node; // 找到目标
338  this->m_hot = node;
339  node = node->child[r+1]; // 原理请看函数说明中的迭代查找示意图
340  // 无论是否找到目标，child[r+1]都为m_hot的子节点
341  }
342  return nullptr;

365 template <typename T>
366 bool BTree<T>::insert(const T& e)
367 {
368  BTNode<T>* node = this->search(e);
369  if (node) return false;
370 
371  // 找到关键码e在m_hot插入的位置
372  int r = this->m_hot->key.search(e);
373  this->m_hot->key.insert(r+1, e);
374  this->m_hot->child.push_back(nullptr); // 因为叶节点所有子节点为nullptr，故插在任意位置均可
375  this->m_size++;
376 
377  this->solve_overflow(this->m_hot);
378 
379  return true;
380 }

389 template <typename T>
390 bool BTree<T>::remove(const T& e)
391 {
392  BTNode<T>* node = this->search(e);
393  if (!node) return false;
394 
395  int r = node->key.search(e);
396  if (node->child[0])
397  {
398  // node不是叶节点，则e右侧必有子节点
399  BTNode<T>* u = node->child[r+1];
400  while(node->child[0]) u = u->child[0]; // 迭代至e的后继(后继必定是叶节点)
401  node->key[r] = u->key[0]; // 将u->key[0]与待删除e交换位置
402  //u->key[0] = e; // 赋不赋值无所谓，反正要删除
403  node = u; // node重新指向包含待删除目标e的叶节点u
404  r = 0; // r也指向e在node->key中的下标
405  }
406 
407  // 如果待删除目标e在叶节点中，直接删除
408  node->key.remove(r);
409  node->child.remove(node->child.size() - 1); // 因为叶节点所有子节点为nullptr，故随便删除一个即可
410  this->m_size--;
411 
412  this->solve_underflow(node);
413  return true;
414 }
