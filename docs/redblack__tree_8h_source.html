131 template <typename T>
132 void RedBlackTree<T>::solve_double_red(BinNodePtr<T> x)
133 {
134  if (BN_IsRoot(*x))
135  {
136  x->color = RBColor::Black;
137  this->m_root->height++; // 变成黑节点，高度加1
138  return;
139  }
140  BinNodePtr<T> p = x->parent;
141  if (BN_IsBlack(p)) // p为黑节点，则没有双红问题
142  return;
143  BinNodePtr<T> g = p->parent; // p为红节点，必有黑父节点
144  BinNodePtr<T> u = BN_IsLeftChild(*p) ? g->right : g->left;
145 
146  if (BN_IsBlack(u))
147  {
148  // RR-1
149  g->color = RBColor::Red;
150  // x,p同侧，即同为左孩子或同为右孩子
151  if (BN_IsLeftChild(*x) == BN_IsLeftChild(*p))
152  p->color = RBColor::Black;
153  else
154  x->color = RBColor::Black;
155 
156  BinNodePtr<T>& n = RefFromParent(*g); // 获取g在父节点的孩子节点指针
157  n = this->rotate_at(x); // 34重构中会更新高度，故重染色需要在rotate_at之前进行
158  //n->color = RBColor::Black;
159  //n->left->color = RBColor::Red;
160  //n->right->color = RBColor::Red;
161  }
162  else
163  {
164  // RR-2 重新染色
165  // u原来为红节点，故一定不为nullptr
166  p->color = RBColor::Black;
167  p->height ++; // 由红转黑，高度加1
168  u->color = RBColor::Black;
169  u->height ++; // 由红转黑，高度加1
170  if (!BN_IsRoot(*g))
171  g->color = RBColor::Red; // 若g不是根节点，需要转红，继续检测双红问题
172  solve_double_red(g); // 若g是根节点，在递归调用中，高度会得到更新
173  }

261 template <typename T>
262 void RedBlackTree<T>::solve_double_black(BinNodePtr<T> r)
263 {
264  // 如果r为nullptr，则r的父节点为m_hot
265  BinNodePtr<T> p = r ? r->parent : this->m_hot;
266  // r为根节点
267  if (!p)
268  return;
269  // r的兄弟节点
270  BinNodePtr<T> s = (r == p->left) ? p->right : p->left;
271 
272  if (BN_IsBlack(s)) // s为黑
273  {
274  BinNodePtr<T> t = nullptr;
275  if (BN_IsRed(s->left)) t = s->left;
276  else if (BN_IsRed(s->right)) t = s->right;
277  if (t) // s至少有一个红子节点
278  {
279  // BB-1
280  RBColor clr = p->color;
281  BinNodePtr<T>& n = RefFromParent(*p);
282  // 因为34重构前，未进行颜色设置，故rotate_at中的高度更新无效
283  n = this->rotate_at(t);
284  // 重染色后，需要重新计算黑高度
285  if (BN_HasLeftChild(*n))
286  {
287  n->left->color = RBColor::Black;
288  this->update_height(n->left);
289  }
290  if (BN_HasRightChild(*n))
291  {
292  n->right->color = RBColor::Black;
293  this->update_height(n->right);
294  }
295  n->color = clr;
296  this->update_height(n);
297  }
298  else // s两个子节点均为黑
299  {
300  s->color = RBColor::Red; // s转红，高度减1
301  s->height--;
302  if (BN_IsRed(p)) // p为红
303  {
304  // BB-2R
305  p->color = RBColor::Black; //p转黑，但黑高度不变
306  }
307  else // p为黑
308  {
309  // BB-2B
310  p->height--; //p保持黑，但黑高度下降
311  solve_double_black(p);
312  }
313  }
314  }
315  else // s为红，两个子节点则一定为黑
316  {
317  // BB-3
318  p->color = RBColor::Red;
319  s->color = RBColor::Black;
320  BinNodePtr<T> t = (BN_IsLeftChild(*s)) ? s->left : s->right;
321  // 取t与s取同侧
322  this->m_hot = p;
323  BinNodePtr<T>& n = RefFromParent(*p);
324  n = this->rotate_at(t);
325  // 递归修正r处的双黑问题
326  solve_double_black(r);
327  }
328 }

380  if (!x)
381  return false;

385  // r为接替x所在位置的节点，r可以为nullptr，m_hot为r的父节点
386  BinNodePtr<T> r = remove_at(x, this->m_hot);
387 
388  // 没有节点了，则直接返回
389  if(!(--this->m_size))
390  return true;
391  // 删除的是根节点，则所有路径少了一个黑节点，则r即为新root节点
392  if(!this->m_hot)
393  {
394  this->m_root->color = RBColor::Black;
395  this->update_height(this->m_root);
396  return true;
397  }
398 
399 #if 1

401  if (xclr == dsa::RBColor::Black && BN_IsRed(r))

406  // x为红，r为黑，则r颜色和高度无需要变化（因为x的黑高度和r的黑高度相等）
407  else if (xclr == dsa::RBColor::Red && BN_IsBlack(r))
408  { }
409  // x为黑，r为黑（x,r不可能同时为红）
