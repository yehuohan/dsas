87 template <typename T>
88 BinNodePtr<T>& SplayTree<T>::search(const T& e)
89 {
90  BinNodePtr<T> p = search_in(this->m_root, e, this->m_hot = nullptr);
91  // 无论是否找到节点，均会进行伸展操作，即没有找到目标节点，也将靠近目标的节点移到树根
92  this->m_root = this->splay(p ? p : this->m_hot);
93  // 若m_root为nullptr（即没有查找到目标e，且m_hot也为nullptr）
94  // 则一定是空树的情况（非空树，即使没有查找到目录，m_hot也不会为nullptr）
95  return this->m_root;
96 }

105 template <typename T>
106 BinNodePtr<T> SplayTree<T>::insert(const T& e)
107 {
108  // 如果是空树，直接插入节点即可
109  if (!this->m_root)
110  {
111  this->m_size++;
112  this->m_root = new BinNode<T>(e, nullptr);
113  return this->m_root;
114  }
115 
116  // 因为不是空树,返回的x不可以为nullptr
117  // search 返回经过伸展后的树根节点
118  if (dsa::is_equal(this->search(e)->data, e)) return this->m_root;
119 
120  BinNodePtr<T> r = this->m_root;
121  this->m_root = new BinNode<T>(e, nullptr);
122  this->m_size++;
123 
124  // 确定新插入节点的位置
125  if (dsa::less_than(e, r->data))
126  {
127  attach_left(this->m_root, r->left);
128  attach_right(this->m_root, r);
129  r->parent = this->m_root;
130  r->left = nullptr;
131 
132  }
133  else if (dsa::less_than(r->data, e))
134  {
135  attach_right(this->m_root, r->right);
136  attach_left(this->m_root, r);
137  r->parent = this->m_root;
138  r->right = nullptr;
139  }
140  this->update_height_above(r);
141 
142  return this->m_root;
143 }

152 template <typename T>
153 bool SplayTree<T>::remove(const T& e)
154 {
155  // 空树直接返回
156  if (!this->m_root) return false;
157  // 没有找到目标
158  if (dsa::not_equal(this->search(e)->data, e)) return false;
159 
160  // 删除目标节点（经过伸展，已经移到了根节点）
161  BinNodePtr<T> s = this->m_root;
162  if (!BN_HasLeftChild(*this->m_root))
163  {
164  this->m_root = this->m_root->right;
165  if (this->m_root) this->m_root->parent = nullptr;
166  }
167  else if (!BN_HasRightChild(*this->m_root))
168  {
169  this->m_root = this->m_root->left;
170  if (this->m_root) this->m_root = nullptr;
171  }
172  else
173  {
174  BinNodePtr<T> lt = this->m_root->left;
175  // 暂时分离左子树
176  this->m_root->left = nullptr;
177  lt->parent = nullptr;
178  // 只保留右子树
179  this->m_root = this->m_root->right;
180  this->m_root->parent = nullptr;
181  // 以原树根为目标，在原右子树做一次（必定失败的）查找，
182  // 原右子树中的最大值经过伸展后，会移到原右子树的根节点
183  this->search(s->data);
184  // 再将原左子树连接到根节点
185  this->m_root->left = lt;
186  lt->parent = this->m_root;
187  }
188 
189  delete s;
190  this->m_size--;
191  if (this->m_root) this->update_height(this->m_root);
192 
193  return true;
194 }

232 template <typename T>
233 BinNodePtr<T> SplayTree<T>::splay(BinNodePtr<T> v)
234 {
235  if (!v) return nullptr;
236  BinNodePtr<T> p;
237  BinNodePtr<T> g;
238 
239  // 双层伸展，伸展完成后，v将成为子树的根节点
240  while((p = v->parent) && (g = p->parent))
241  {
242  BinNodePtr<T> gg = g->parent;
243  if (BN_IsLeftChild(*v))
244  {
245  if (BN_IsLeftChild(*p))
246  {
247  // zig-zig
248  attach_left(p, v->right);
249  attach_left(g, p->right);
250  attach_right(v, p);
251  attach_right(p, g);
252  }
253  else
254  {
255  // zig-zag
256  this->connect34(g, v, p,
257  g->left, v->left, v->right, p->right);
258  }
259  }
260  else
261  {
262  if (BN_IsRightChild(*p))
263  {
264  // zag-zag
265  attach_right(p, v->left);
266  attach_right(g, p->left);
267  attach_left(v, p);
268  attach_left(p, g);
269  }
270  else
271  {
272  // zag-zig
273  this->connect34(g, v, g,
274  p->left, v->left, v->right, g->right);
275  }
276 
277  }
278 
279  // 将新的子树接入到原树的位置
280  if (!gg)
281  v->parent = nullptr; // 成为树根节点
282  else
283  (g == gg->left) ? attach_left(gg, v) : attach_right(gg, v);
284  // 插入到原树作为左子树或右子树
285 
286  this->update_height(g);
287  this->update_height(p);
288  this->update_height(v);
289  }
290 
291  // 单层伸展，最多一次旋转
292  if ((p = v->parent))
293  {
294  if (BN_IsLeftChild(*v))
295  {
296  attach_left(p, v->right);
297  attach_right(v, p);
298  }
299  else
300  {
301  attach_right(p, v->left);
302  attach_left(v, p);
303  }
304  }
305 
306  // v成为树的根节点
307  v->parent = nullptr;
308  return v;
309 }
