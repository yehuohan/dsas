243 int match_bf1(char* P, char* T)
244 {
245  int n = str_len(T), i = 0;
246  int m = str_len(P), j = 0;
247 
248  while(j < m && i < n)
249  {
250  if (T[i] == P[j])
251  {
252  i++;
253  j++;
254  }
255  else
256  {
257  i -= (j-1); // T回退，即回退了 j 个字符，再前进 1 个字符
258  j = 0; // P复位，即回退了 j 个字符
259  }
260  }
261  return i-j; // 若匹配成功，则i-j为P在T首次出现位置（第一个字符）的下标
262 }

390  }
391  else

393  }

395  return i-j; // 若匹配成功，则i-j为P在T首次出现位置（第一个字符）的下标

662 int* build_bc(char* P)
663 {
664  // char类型最多0xFF个
665  int* bc = new int[256];
666 
667  for (int k = 0; k < 256; k++)
668  bc[k] = -1;
669  for (int m = str_len(P), j = 0; j < m; j++)
670  bc[(unsigned int)P[j]] = j; // 覆盖刷新字符P[j]的出现位置记录，即重复的字符，下标必定是最大的
671  // 且没有的字符均为-1
672  return bc;

752 int* build_gs(char* P)
753 {
754  int m = str_len(P);
755  int* gs = new int[m];
756  int* ss = new int[m];
757 
758 #if(0)
759  // build ss，蛮力计算 O(m^2)
760  for (int k = m-1; k >= 0; k--)
761  {
762  int j = k;
763  for (int n = m-1; n >= 0; n--)
764  {
765  if(P[n] != P[j])
766  {
767  // P的子串和后缀相等
768  ss[k] = k-j;
769  break;
770  }
771  if (--j < 0)
772  {
773  // P的前缀和后缀相等
774  ss[k] = k-j;
775  break;
776  }
777  }
778  }
779 #else
780  // 优化算法 O(m)
781  ss[m-1] = m; //对最后一个字符而言，与之匹配的最长后缀就是整个P串
782  for (int lo = m - 1, hi = m - 1, j = lo - 1; j >= 0; j --)
783  {
784  if (( lo < j) &&
785  (ss[m - (hi-j+1)] <= j - lo))
786  //情况一
787  ss[j] = ss[m - (hi-j+1)]; // 利用此前已计算出的ss[]
788  else
789  {
790  //情况二
791  hi = j;
792  lo = (lo < hi) ? lo : hi;
793  while ((0 <= lo) &&
794  (P[lo] == P[m - (hi-lo+1)]))
795  lo--; // 逐个对比处位于(lo, hi]前端的字符
796  ss[j] = hi - lo;
797  }
798  }
799 #endif
800 
801  // build gs
802  for (int k = 0; k < m; k ++)
803  gs[k] = m;
804  for (int i = 0, k = m-1; k >= 0; k--)
805  {
806  // (2.1)的情况
807  // 因为k是倒序遍历，所在先遇到的是较长的前缀，因而先计算ss的也是P中较前面的字符
808  if (ss[k] == k + 1)
809  {
810  while (i < m-k-1)
811  gs[i++] = m-(k+1);
812  }
813  }
814  for (int k = 0; k < m-1; k++)
815  gs[m-ss[k]-1] = m-k-1; // (2.2)的情况，不会影响(2.1)情况中计算的值
816 
817  delete[] ss;
818  return gs;
819 }
