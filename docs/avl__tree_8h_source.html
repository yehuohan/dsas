109 BinNode<T>* AvlTree<T>::insert(const T& e)
110 {
111  BinNode<T>*& x = this->search(e);
112  if (x) return x; // 不插入重复节点
113 
114  x = new BinNode<T>(e, this->m_hot); // 新插入的节点
115  this->m_size ++;
116  BinNode<T>* node = x;
117 
118  // 从新插入节点的父节点开始，向上遍历所有父节点，检查是否满足Avl平衡
119  for(BinNode<T>* g = this->m_hot; g; g = g->parent)
120  {
121  if(!AVL_Balanced(*g)) // 所有失衡节点中的最低点
122  {
123  BinNode<T>*& sub_node = RefFromParent(*g);
124  // 获取g在父节点的孩子节点指针
125  sub_node = this->rotate_at(AVL_TallerChild(AVL_TallerChild(g)));
126  // 将rotate_at返回子树的根节点添加到Avl树中
127  break;
128  }
129  else
130  this->update_height(g);
131  }
132 
133  return node;
134 }

143 template <typename T>
144 bool AvlTree<T>::remove(const T& e)
145 {
146  BinNode<T>*& x = this->search(e);
147  if (!x) return false;
148  remove_at(x, this->m_hot);
149  this->m_size--;
150 
151  for (BinNode<T>* g = this->m_hot; g; g = g->parent)
152  {
153  if (!AVL_Balanced(*g))
154  {
155  BinNode<T>*& sub_node = RefFromParent(*g);
156  g = sub_node = this->rotate_at(AVL_TallerChild(AVL_TallerChild(g)));
157  }
158  this->update_height(g);
159  }
160 
161  return true;
162 }
