162 template <typename T>
163 BinNode<T>* BinNode<T>::successor()
164 {
165  BinNode<T>* s = this;
166  if (this->right)
167  {
168  s = this->right;
169  while(s->left) s = s->left; //为右子树中最靠左（最小）的节点
170  }
171  else
172  {
173  //或者为“将当前节点包含于其左子树中的最低祖先”
174  if (s->parent && s->parent->right == s) s = s->parent;
175  s = s->parent;
176  }
177  return s;
178 }

341 template <typename T, typename VST>
342 void traverse_LRD_iteration(BinNode<T>* node, VST& visit)
343 {
344  dsa::Stack<BinNode<T>*> s;
345  BinNode<T>* last_node = nullptr;
346 
347  // 先将左子树节点均入栈
348  go_along_left_branch(node, s);
349  while(!s.is_empty())
350  {
351  node = s.pop();
352  // LRD中的D节点能被访问的前提是：无右子树或右子树已被访问过
353  if(!node->right || node->right == last_node)
354  {
355  visit(node->data);
356  last_node = node;
357  }
358  else
359  {
360  // 将LRD中的D重新入栈，等待右子树被访问完
361  s.push(node);
362  node = node->right;
363  go_along_left_branch(node, s);
364  }
365  }
366 }
