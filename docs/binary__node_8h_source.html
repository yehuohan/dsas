343 template <typename T, typename VST>
344 void traverse_LRD_iteration(BinNodePtr<T> node, VST& visit)
345 {
346  dsa::Stack<BinNodePtr<T> > s;
347  BinNodePtr<T> last_node = nullptr;
348 
349  // 先将左子树节点均入栈
350  go_along_left_branch(node, s);
351  while(!s.is_empty())
352  {
353  node = s.pop();
354  // LRD中的D节点能被访问的前提是：无右子树或右子树已被访问过
355  if(!node->right || node->right == last_node)
356  {
357  visit(node->data);
358  last_node = node;
359  }
360  else
361  {
362  // 将LRD中的D重新入栈，等待右子树被访问完
363  s.push(node);
364  node = node->right;
365  go_along_left_branch(node, s);
366  }
367  }
368 }
