169 template <typename T>
170 BinNode<T>* BinNode<T>::successor()
171 {
172  BinNode<T>* s = this;
173  if (this->right)
174  {
175  s = this->right;
176  while(s->left) s = s->left; //为右子树中最靠左（最小）的节点
177  }
178  else
179  {
180  //或者为“将当前节点包含于其左子树中的最低祖先”
181  if (s->parent && s->parent->right == s) s = s->parent;
182  s = s->parent;
183  }
184  return s;
185 }

294 {

348 template <typename T, typename VST>
349 void traverse_LRD_iteration(BinNode<T>* node, VST& visit)
350 {
351  dsa::Stack<BinNode<T>*> s;
352  BinNode<T>* last_node = nullptr;
353 
354  // 先将左子树节点均入栈
355  go_along_left_branch(node, s);
356  while(!s.is_empty())
357  {
358  node = s.pop();
359  // LRD中的D节点能被访问的前提是：无右子树或右子树已被访问过
360  if(!node->right || node->right == last_node)
361  {
362  visit(node->data);
363  last_node = node;
364  }
365  else
366  {
367  // 将LRD中的D重新入栈，等待右子树被访问完
368  s.push(node);
369  node = node->right;
370  go_along_left_branch(node, s);
371  }
372  }
373 }
