492 void Vector<T>::merge(int lo, int mi, int hi)
493 {
494  /* merge [lo, mi) and [mi, hi) */
495  T* p = this->m_ar + lo;
496  int len = mi - lo;
497  T* left = new T[len];
498  for(int k = 0; k < len; k++)
499  {
500  *(left + k) = *(p + k);
501  }
502  T* right = this->m_ar + mi;
503 
504  int i = 0, j = 0, k = 0;
505  while(i < mi-lo && j < hi-mi)
506  {

509  p[k++] = left[i++];
510  }

513  p[k++] = right[j++];
514  }
515  }

519  }
