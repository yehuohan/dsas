591 template <typename T>
592 void Vector<T>::merge(int lo, int mi, int hi)
593 {
594  T* p = this->m_array + lo;
595  int len = mi - lo;
596  T* left = new T[len];
597  for(int k = 0; k < len; k++)
598  left[k] = p[k];
599  T* right = this->m_array + mi;
600 
601  int i = 0, j = 0, k = 0;
602  while(i < mi-lo && j < hi-mi)
603  {
604  if(left[i] <= right[j])
605  p[k++] = left[i++];
606  else
607  p[k++] = right[j++];
608  }
609  while(i < mi - lo) p[k++] = left[i++];
610  while(j < hi - mi) p[k++] = right[j++];
611  delete[] left;
612 }
