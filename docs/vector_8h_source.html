502 void Vector<T>::merge(int lo, int mi, int hi)
503 {
504  /* merge [lo, mi) and [mi, hi) */
505  T* p = this->m_ar + lo;
506  int len = mi - lo;
507  T* left = new T[len];
508  for(int k = 0; k < len; k++)
509  {
510  *(left + k) = *(p + k);
511  }
512  T* right = this->m_ar + mi;
513 
514  int i = 0, j = 0, k = 0;
515  while(i < mi-lo && j < hi-mi)
516  {
517  if(left[i] <= right[j])
518  {
519  p[k++] = left[i++];
520  }
521  else
522  {
523  p[k++] = right[j++];
524  }
525  }
526  while(i < mi -lo)

528  p[k++] = left[i++];
529  }
530  while(j < hi - mi)

532  p[k++] = right[j++];
533  }
