925 template <typename T>
926 void Vector<T>::shell_insertion(int w, int lo, int hi)
927 {
928 #if(1)
929  // 每w次循环，完成对所有列的同一行的插入排序
930  // 第一行元素： lo , lo+1 , ... lo+w-1
931  // 第二行元素： lo+w , lo+w+1 , ...
932  for (int k = lo + w; k < hi; k ++)
933  {
934  // 此部分程序参照insertion_sort
935  int n = k;
936  T tmp = this->m_array[n];
937  while (n >= lo+w && tmp < this->m_array[n-w])
938  {
939  this->m_array[n] = this->m_array[n-w];
940  n -= w;
941  }
942  this->m_array[n] = tmp;
943  }
944 
945 #else
946  // 依次对第 0,1,2...w 列进行排序(排完第0列，才能排第1列······)
947  int cow = ((hi-lo) + w - 1) / w; // 向上取整，即为矩阵的行数
948  for (int k = 0; k < w; k ++) // 对所有列进行插入排序
949  {
950  int sl = lo + k, sh = lo + k+cow*w; // (插入排序)列下标范围[sl,sh)
951  if (sh > hi) sh -= w; // 防止列下标超范围
952  for (int s = sl + w; s < sh; s += w)// 对列元素进行插入排序
953  {
954  // 此部分程序参照insertion_sort
955  int n = s;
956  T tmp = this->m_array[n];
957  while (n > sl && tmp < this->m_array[n-w])
958  {
959  this->m_array[n] = this->m_array[n-w];
960  n -= w;
961  }
962  this->m_array[n] = tmp;
963  }
964  }
965 #endif
966 }
