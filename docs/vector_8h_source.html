896 template <typename T>
897 void Vector<T>::shell_insertion(int w, int lo, int hi)
898 {
899 #if(1)
900  // 每w次循环，完成对所有列的同一行的插入排序
901  // 第一行元素： lo , lo+1 , ... lo+w-1
902  // 第二行元素： lo+w , lo+w+1 , ...
903  for (int k = lo + w; k < hi; k ++)
904  {
905  // 此部分程序参照insertion_sort
906  int n = k;
907  T tmp = this->m_array[n];
908  while (n >= lo+w && tmp < this->m_array[n-w])
909  {
910  this->m_array[n] = this->m_array[n-w];
911  n -= w;
912  }
913  this->m_array[n] = tmp;
914  }
915 
916 #else
917  // 依次对第 0,1,2...w 列进行排序(排完第0列，才能排第1列······)
918  int cow = ((hi-lo) + w - 1) / w; // 向上取整，即为矩阵的行数
919  for (int k = 0; k < w; k ++) // 对所有列进行插入排序
920  {
921  int sl = lo + k, sh = lo + k+cow*w; // (插入排序)列下标范围[sl,sh)
922  if (sh > hi) sh -= w; // 防止列下标超范围
923  for (int s = sl + w; s < sh; s += w)// 对列元素进行插入排序
924  {
925  // 此部分程序参照insertion_sort
926  int n = s;
927  T tmp = this->m_array[n];
928  while (n > sl && tmp < this->m_array[n-w])
929  {
930  this->m_array[n] = this->m_array[n-w];
931  n -= w;
932  }
933  this->m_array[n] = tmp;
934  }
935  }
936 #endif
937 }
