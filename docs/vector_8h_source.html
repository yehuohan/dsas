929 template <typename T>
930 void Vector<T>::shell_insertion(int w, int lo, int hi)
931 {
932 #if(1)
933  // 每w次循环，完成对所有列的同一行的插入排序
934  // 第一行元素： lo , lo+1 , ... lo+w-1
935  // 第二行元素： lo+w , lo+w+1 , ...
936  for (int k = lo + w; k < hi; k ++)
937  {
938  // 此部分程序参照insertion_sort
939  int n = k;
940  T tmp = this->m_array[n];
941  while (n >= lo+w && tmp < this->m_array[n-w])
942  {
943  this->m_array[n] = this->m_array[n-w];
944  n -= w;
945  }
946  this->m_array[n] = tmp;
947  }
948 
949 #else
950  // 依次对第 0,1,2...w 列进行排序(排完第0列，才能排第1列······)
951  int cow = ((hi-lo) + w - 1) / w; // 向上取整，即为矩阵的行数
952  for (int k = 0; k < w; k ++) // 对所有列进行插入排序
953  {
954  int sl = lo + k, sh = lo + k+cow*w; // (插入排序)列下标范围[sl,sh)
955  if (sh > hi) sh -= w; // 防止列下标超范围
956  for (int s = sl + w; s < sh; s += w)// 对列元素进行插入排序
957  {
958  // 此部分程序参照insertion_sort
959  int n = s;
960  T tmp = this->m_array[n];
961  while (n > sl && tmp < this->m_array[n-w])
962  {
963  this->m_array[n] = this->m_array[n-w];
964  n -= w;
965  }
966  this->m_array[n] = tmp;
967  }
968  }
969 #endif
