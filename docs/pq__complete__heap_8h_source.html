121 template <typename T>
122 int PqComplHeap<T>::proper_parent(int n, int i)
123 {
124 #define PQ_Bigger(i, j) (((this->m_ar[i]) >= (this->m_ar[j])) ? (i) : (j))
125  if (PQ_RightChildValid(n, i))
126  {
127  // i的左右孩子均存在
128  return PQ_Bigger(i, PQ_Bigger(PQ_LeftChild(i), PQ_RightChild(i)) );
129  }
130  else if(PQ_LeftChildValid(n, i))
131  {
132  // 只有i的左孩子存在
133  return PQ_Bigger(i, PQ_LeftChild(i));
134  }
135  else
136  return i;
137 #undef PQ_Bigger
138 }

147 template <typename T>
148 int PqComplHeap<T>::percolate_up(int i)
149 {
150  T ele = this->m_ar[i];
151  while(PQ_ParentValid(i))
152  {
153  int j = PQ_Parent(i);
154  if (ele < this->m_ar[j])
155  break;
156  // 新插入元素比父元素大，则上滤元素
157  this->m_ar[i] = this->m_ar[j];
158  i = j;
159  }
160  // 将新插入元素，放到最终的节点位置
161  this->m_ar[i] = ele;
162  return i;
163 }

175 template <typename T>
176 int PqComplHeap<T>::percolate_down(int n, int i)
177 {
178  int j;
179  while(i!= (j = this->proper_parent(n, i)))
180  {
181  // i的子节点元比i大，则下滤
182  T tmp = this->m_ar[i];
183  this->m_ar[i] = this->m_ar[j];
184  this->m_ar[j] = tmp;
185  i = j;
186  }
187  return i;
188 }

244 template <typename T>

246 {
247 #if(0)
248  // 按自上而下的顺序上滤
249  // 效率太低：对所有节点的深度求和
250  // 越往堆底，深度越大，节点数越多
251  // Sum[depth(i)] = O(nlog(n))
252  for (int k = 1; k < n; k ++)
253  {
254  this->percolate_up(k);
255  }
256 #endif

259  // 按自下而上的顺序下滤
260  // PQ_Parent(n-1) ：最后一个内部节点，也即末节点的父亲

262  // 越往堆底，高度越小，节点数越多
263  // Sum[height(i)] = O(n)
264  for(int k = PQ_Parent(n-1); k >= 0; k--)
265  {
266  this->percolate_down(n, k);
