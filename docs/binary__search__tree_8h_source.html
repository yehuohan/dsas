143 template <typename T>
144 static BinNodePtr<T> remove_at(
145  BinNodePtr<T>& node,
146  BinNodePtr<T>& hot)
147 {
148  BinNodePtr<T> w = node;
149  BinNodePtr<T> succ = nullptr;
150  // 只有右子树(或左右子树均没有，则返回nullptr，即被删除节点位置没有新节点)
151  if(!node->left) succ = node = node->right;
152  // 只有左子树
153  else if(!node->right) succ = node = node->left;
154  // 左右分支均存在
155  else
156  {
157  // 右子树存在，故node的直接后继w为node右子树最靠左（最小）的节点
158  w = w->successor();
159  // 交换直接后继w与目标节点的数据，交换完后，w成为待删除的目标节点
160  dsa::swap(w->data, node->data);
161  BinNodePtr<T> u = w->parent;
162  // 后继节点w只可能有右子树，不可能有左子树
163  if (u == node)
164  {
165  // 后继节点w即为node的右子节点
166  u->right = succ = w->right;
167  }
168  else
169  {
170  u->left = succ = w->right;
171  }
172  }
173 
174  hot = w->parent; // 待删除节点的父节点
175  // succ不为nullptr，则hot为succ的父节点
176  // succ为nullptr，则hot的子节点w被删除后，也为nullptr
177  if (succ) succ->parent = hot; // 设置接替节点的父节点
178  delete w; // 删除节点
179 
180  return succ;
181 }

277 template <typename T>
278 BinNodePtr<T> BinSearchTree<T>::rotate_at(BinNodePtr<T> v)
279 {
280  BinNodePtr<T> p = v->parent;
281  BinNodePtr<T> g = p->parent;
282  if (BN_IsLeftChild(*p))
283  {
284  if (BN_IsLeftChild(*v))
285  {
286  // zig 单旋
287  p->parent = g->parent; // 设置子树根节点的父节点
288  return this->connect34(v, p, g,
289  v->left, v->right, p->right, g->right);
290  }
291  else
292  {
293  // zig-zag 双旋
294  v->parent = g->parent; // 设置子树根节点的父节点
295  return this->connect34(p, v, g,
296  p->left, v->left, v->right, g->right);
297  }
298  }
299  else if (BN_IsRightChild(*p))
300  {
301  if (BN_IsLeftChild(*v))
302  {
303  // zag-zig 双旋
304  v->parent = g->parent; // 设置子树根节点的父节点
305  return this->connect34(g, v, p,
306  g->left, v->left, v->right, p->right);
307  }
308  else
309  {
310  // zag 单旋
311  p->parent = g->parent; // 设置子树根节点的父节点
312  return this->connect34(g, p, v,
313  g->left, p->left, v->left, v->right);
