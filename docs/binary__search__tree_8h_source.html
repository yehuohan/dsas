141 template <typename T>
142 static BinNode<T>* remove_at(
143  BinNode<T>*& node,
144  BinNode<T>*& hot)
145 {
146  BinNode<T>* w = node;
147  BinNode<T>* succ = nullptr;
148  // 只有右子树(或左右子树均没有，则返回nullptr，即被删除节点位置没有新节点)
149  if(!node->left) succ = node = node->right;
150  // 只有左子树
151  else if(!node->right) succ = node = node->left;
152  // 左右分支均存在
153  else
154  {
155  // 右子树存在，故node的直接后继w为node右子树最靠左（最小）的节点
156  w = w->successor();
157  // 交换直接后继w与目标节点的数据，交换完后，w成为待删除的目标节点
158  T tmp = w->data;
159  w->data = node->data;
160  node->data = tmp;
161  BinNode<T>* u = w->parent;
162  // 后继节点w只可能有右子树，不可能有左子树
163  if (u == node)
164  {
165  // 后继节点w即为node的右子节点
166  u->right = succ = w->right;
167  }
168  else
169  {
170  u->left = succ = w->right;
171  }
172  }
173 
174  hot = w->parent; // 待删除节点的父节点
175  // succ不为nullptr，则hot为succ的父节点
176  // succ为nullptr，则hot的子节点w被删除后，也为nullptr
177  if (succ) succ->parent = hot; // 设置接替节点的父节点
178  delete w; // 删除节点
179 
180  return succ;
181 }

274 template <typename T>
275 BinNode<T>* BinSearchTree<T>::rotate_at(BinNode<T>* v)
276 {
277  BinNode<T>* p = v->parent;
278  BinNode<T>* g = p->parent;
279  if (BN_IsLeftChild(*p))
280  {
281  if (BN_IsLeftChild(*v))
282  {
283  // zig 单旋
284  p->parent = g->parent; // 设置子树根节点的父节点
285  return this->connect34(v, p, g,
286  v->left, v->right, p->right, g->right);
287  }
288  else
289  {
290  // zig-zag 双旋
291  v->parent = g->parent; // 设置子树根节点的父节点
292  return this->connect34(p, v, g,
293  p->left, v->left, v->right, g->right);
294  }
295  }
296  else if (BN_IsRightChild(*p))
297  {
298  if (BN_IsLeftChild(*v))
299  {
300  // zag-zig 双旋
301  v->parent = g->parent; // 设置子树根节点的父节点
302  return this->connect34(g, v, p,
303  g->left, v->left, v->right, p->right);
304  }
305  else
306  {
307  // zag 单旋
308  p->parent = g->parent; // 设置子树根节点的父节点
309  return this->connect34(g, p, v,
310  g->left, p->left, v->left, v->right);
311  }
312  }
313 }
