158 template <typename T>
159 static BinNode<T>* remove_at(
160  BinNode<T>*& node,
161  BinNode<T>*& hot)
162 {
163  BinNode<T>* w = node;
164  BinNode<T>* succ = nullptr;
165  // 只有右子树(或左右子树均没有，则返回nullptr，即被删除节点位置没有新节点)
166  if(!node->left) succ = node = node->right;
167  // 只有左子树
168  else if(!node->right) succ = node = node->left;
169  // 左右分支均存在
170  else
171  {
172  // 右子树存在，故node的直接后继w为node右子树最靠左（最小）的节点
173  w = w->successor();
174  // 交换直接后继w与目标节点的数据，交换完后，w成为待删除的目标节点
175  T tmp = w->data;
176  w->data = node->data;
177  node->data = tmp;
178  BinNode<T>* u = w->parent;
179  // 后继节点w只可能有右子树，不可能有左子树
180  if (u == node)
181  {
182  // 后继节点w即为node的右子节点
183  u->right = succ = w->right;
184  }
185  else
186  {
187  u->left = succ = w->right;
188  }
189  }
190 
191  hot = w->parent; // 待删除节点的父节点
192  // succ不为nullptr，则hot为succ的父节点
193  // succ为nullptr，则hot的子节点w被删除后，也为nullptr
194  if (succ) succ->parent = hot; // 设置接替节点的父节点
195  delete w; // 删除节点
196 
197  return succ;
198 }

291 template <typename T>
292 BinNode<T>* BinSearchTree<T>::rotate_at(BinNode<T>* v)
293 {
294  BinNode<T>* p = v->parent;
295  BinNode<T>* g = p->parent;
296  if (BN_IsLeftChild(*p))
297  {
298  if (BN_IsLeftChild(*v))
299  {
300  // zig 单旋
301  p->parent = g->parent; // 设置子树根节点的父节点
302  return this->connect34(v, p, g,
303  v->left, v->right, p->right, g->right);
304  }
305  else
306  {
307  // zig-zag 双旋
308  v->parent = g->parent; // 设置子树根节点的父节点
309  return this->connect34(p, v, g,
310  p->left, v->left, v->right, g->right);
311  }
312  }
313  else if (BN_IsRightChild(*p))
314  {
315  if (BN_IsLeftChild(*v))
316  {
317  // zag-zig 双旋
318  v->parent = g->parent; // 设置子树根节点的父节点
319  return this->connect34(g, v, p,
320  g->left, v->left, v->right, p->right);
321  }
322  else
323  {
324  // zag 单旋
325  p->parent = g->parent; // 设置子树根节点的父节点
326  return this->connect34(g, p, v,
327  g->left, p->left, v->left, v->right);
328  }
329  }
330 }
