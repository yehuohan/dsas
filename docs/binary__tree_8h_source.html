67 template <typename T>
68 int BinTree<T>::update_height(BinNode<T>* node)
69 {
70  // 叶子结点没有子结点了，其高度为 1 + (-1) = 0
71  int a = node->left ? node->left->height : -1;
72  int b = node->right ? node->right->height : -1;
73  return node->height = 1 + ((a>=b)?a:b);
74 }

117 template <typename T>
118 int BinTree<T>::remove(BinNode<T>* node)
119 {
120  BinNode<T>* p = node->parent;
121  if (p)
122  {
123  if (p->left == node) p->left = nullptr;
124  else p->right = nullptr;
125  }
126  this->update_height_above(p);
127  int n = remove_at(node);
128  this->m_size -= n;
129  return n;

176 template <typename T>
177 static void construct_bintree_pre_in(const dsa::Vector<T>& pre, const dsa::Vector<T>& in, dsa::BinNode<T>*& node)
178 {
179  // 查找子树的根节点
180  int d_index = in.find(pre[0], 0, in.size());
181  dsa::Vector<T> l_pre, l_in; // 左子树先序与中序序列
182  dsa::Vector<T> r_pre, r_in; // 右子树先序与中序序列
183 
184  // 转移子树的先序与中序，先序与中序长度一样
185  for(int k = 0; k < d_index; k ++)
186  {
187  l_pre.push_back(pre[k+1]); // 转移左子树先序
188  l_in.push_back(in[k]); // 转移左子树中序
189  }
190  for(int k = d_index+1; k < in.size(); k++)
191  {
192  r_pre.push_back(pre[k]); // 转移右子树先序
193  r_in.push_back(in[k]); // 转移右子树中序
194  }
195 
196  // 若子树为空，则不需要继续向下构建
197  if(!l_pre.is_empty())
198  {
199  // 插件子树的根节点
200  dsa::BinNode<T>* left = node->insert_left(l_pre[0]);
201  // 继续重构左子树与右子树
202  construct_bintree_pre_in(l_pre, l_in, left);
203  }
204  if(!r_pre.is_empty())
205  {
206  dsa::BinNode<T>* right = node->insert_right(r_pre[0]);
207  construct_bintree_pre_in(r_pre, r_in, right);
208  }
209 }
