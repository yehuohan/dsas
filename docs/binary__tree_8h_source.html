108 template <typename T>
109 int BinTree<T>::update_height(BinNodePtr<T> node)
110 {
111  // 叶子结点没有子结点了，其高度为 1 + (-1) = 0
112  int a = node->left ? node->left->height : -1;
113  int b = node->right ? node->right->height : -1;
114  return node->height = 1 + ((a>=b)?a:b);
115 }

218 static void construct_bintree_pre_in(
219  const dsa::Vector<T>& pre,
220  const dsa::Vector<T>& in,
221  BinNodePtr<T>& node)
222 {
223  // 查找子树的根节点
224  int d_index = in.find(pre[0], 0, in.size());
225  dsa::Vector<T> l_pre, l_in; // 左子树先序与中序序列
226  dsa::Vector<T> r_pre, r_in; // 右子树先序与中序序列
227 
228  // 转移子树的先序与中序，先序与中序长度一样
229  for(int k = 0; k < d_index; k ++)
230  {
231  l_pre.push_back(pre[k+1]); // 转移左子树先序
232  l_in.push_back(in[k]); // 转移左子树中序
233  }
234  for(int k = d_index+1; k < in.size(); k++)
235  {
236  r_pre.push_back(pre[k]); // 转移右子树先序
237  r_in.push_back(in[k]); // 转移右子树中序
238  }
239 
240  // 若子树为空，则不需要继续向下构建
241  if(!l_pre.is_empty())
242  {
243  // 插件子树的根节点
244  BinNodePtr<T> left = node->insert_left(l_pre[0]);
245  // 继续重构左子树与右子树
246  construct_bintree_pre_in(l_pre, l_in, left);
247  }
248  if(!r_pre.is_empty())
249  {
250  BinNodePtr<T> right = node->insert_right(r_pre[0]);
251  construct_bintree_pre_in(r_pre, r_in, right);
252  }
253 }
