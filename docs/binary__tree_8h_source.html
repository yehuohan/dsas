108 template <typename T>
109 int BinTree<T>::update_height(BinNode<T>* node)
110 {
111  // 叶子结点没有子结点了，其高度为 1 + (-1) = 0
112  int a = node->left ? node->left->height : -1;
113  int b = node->right ? node->right->height : -1;
114  return node->height = 1 + ((a>=b)?a:b);
115 }

159 int BinTree<T>::remove(BinNode<T>* node)
160 {
161  BinNode<T>* p = node->parent;
162  if (p)
163  {
164  if (p->left == node) p->left = nullptr;
165  else p->right = nullptr;
166  }
167  this->update_height_above(p);
168  int n = remove_at(node);
169  this->m_size -= n;
170  return n;

217 template <typename T>
218 static void construct_bintree_pre_in(const dsa::Vector<T>& pre, const dsa::Vector<T>& in, dsa::BinNode<T>*& node)
219 {
220  // 查找子树的根节点
221  int d_index = in.find(pre[0], 0, in.size());
222  dsa::Vector<T> l_pre, l_in; // 左子树先序与中序序列
223  dsa::Vector<T> r_pre, r_in; // 右子树先序与中序序列
224 
225  // 转移子树的先序与中序，先序与中序长度一样
226  for(int k = 0; k < d_index; k ++)
227  {
228  l_pre.push_back(pre[k+1]); // 转移左子树先序
229  l_in.push_back(in[k]); // 转移左子树中序
230  }
231  for(int k = d_index+1; k < in.size(); k++)
232  {
233  r_pre.push_back(pre[k]); // 转移右子树先序
234  r_in.push_back(in[k]); // 转移右子树中序
235  }
236 
237  // 若子树为空，则不需要继续向下构建
238  if(!l_pre.is_empty())
239  {
240  // 插件子树的根节点
241  dsa::BinNode<T>* left = node->insert_left(l_pre[0]);
242  // 继续重构左子树与右子树
243  construct_bintree_pre_in(l_pre, l_in, left);
244  }
245  if(!r_pre.is_empty())
246  {
247  dsa::BinNode<T>* right = node->insert_right(r_pre[0]);
248  construct_bintree_pre_in(r_pre, r_in, right);
249  }
250 }
