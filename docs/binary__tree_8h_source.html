106 template <typename T>
107 int BinTree<T>::update_height(BinNodePtr<T> node)
108 {
109  // 叶子结点没有子结点了，其高度为 1 + (-1) = 0
110  int a = node->left ? node->left->height : -1;
111  int b = node->right ? node->right->height : -1;
112  return node->height = 1 + ((a>=b)?a:b);
113 }

216 static void construct_bintree_pre_in(
217  const dsa::Vector<T>& pre,
218  const dsa::Vector<T>& in,
219  BinNodePtr<T>& node)
220 {
221  // 查找子树的根节点
222  int d_index = in.find(pre[0], 0, in.size());
223  dsa::Vector<T> l_pre, l_in; // 左子树先序与中序序列
224  dsa::Vector<T> r_pre, r_in; // 右子树先序与中序序列
225 
226  // 转移子树的先序与中序，先序与中序长度一样
227  for(int k = 0; k < d_index; k ++)
228  {
229  l_pre.push_back(pre[k+1]); // 转移左子树先序
230  l_in.push_back(in[k]); // 转移左子树中序
231  }
232  for(int k = d_index+1; k < in.size(); k++)
233  {
234  r_pre.push_back(pre[k]); // 转移右子树先序
235  r_in.push_back(in[k]); // 转移右子树中序
236  }
237 
238  // 若子树为空，则不需要继续向下构建
239  if(!l_pre.is_empty())
240  {
241  // 插件子树的根节点
242  BinNodePtr<T> left = node->insert_left(l_pre[0]);
243  // 继续重构左子树与右子树
244  construct_bintree_pre_in(l_pre, l_in, left);
245  }
246  if(!r_pre.is_empty())
247  {
248  BinNodePtr<T> right = node->insert_right(r_pre[0]);
249  construct_bintree_pre_in(r_pre, r_in, right);
250  }
251 }
