201 template <typename Tv, typename Te>
202 Tv GraphMatrix<Tv,Te>::remove_vertex(int i)
203 {
204  for(int j = 0; j < this->m_vnum; j++)
205  {
206  // 删除顶点的出边
207  if (this->exist_edge(i,j))
208  {
209  delete this->m_E[i][j];
210  // 头部(终点)顶点 j 的入度减1
211  this->m_V[j]->in_deg--;
212  }
213  }
214  // 删除第i行
215  this->m_E.remove(i);
216  this->m_vnum--;
217 
218  for(int j = 0; j < this->m_vnum; j++)
219  {
220  // 删除顶点的入边
221  if (this->exist_edge(j,i))
222  {
223  // 删除边，并且每行删除对应顶点 i 的元素
224  delete this->m_E[j].remove(i);
225  // 尾部(起点)顶点 j 的出度减1
226  this->m_V[j]->out_deg--;
227  }
228  }
229 
230  Tv old = this->m_V[i]->data;
231  delete this->m_V.remove(i);
232  return old;
233 }

245 template <typename Tv, typename Te>
246 void GraphMatrix<Tv,Te>::BFS(int vindex, int& clock)
247 {
248  this->m_V[vindex]->status = VStatus::Discovered;
249 
250  dsa::Queue<int> q;
251  q.enqueue(vindex);
252  while(!q.is_empty())
253  {
254  int v = q.dequeue();
255  this->m_V[v]->d_time = ++clock;
256 
257  for (int k = this->first_nbr(v); k > -1 ; k = this->next_nbr(v, k))
258  {
259  if (VStatus::UnDiscovered == this->m_V[k]->status)
260  {
261  this->m_V[k]->status = VStatus::Discovered;
262  q.enqueue(k); // 在队列中的顶点处理Discovered状态
263  this->m_V[k]->parent = v;
264  this->m_E[v][k]->status = EStatus::Tree;
265  // 将边引入Tree中（处于Discovered均为引入Tree中）
266  }
267  else
268  {
269  this->m_E[v][k]->status = EStatus::Cross;
270  // 若已在队列中，或是已经访问完毕，则将边引入Cross边中
271  }
272  }
273 
274  this->m_V[v]->status = VStatus::Visited;
275  }
276 }

286 template <typename Tv, typename Te>
287 void GraphMatrix<Tv,Te>::DFS(int vindex, int& clock)
288 {
289  this->m_V[vindex]->status = VStatus::Discovered;
290  this->m_V[vindex]->d_time = ++clock;
291 
292  for (int k = this->first_nbr(vindex); k > -1 ; k = this->next_nbr(vindex, k))
293  {
294  switch(this->m_V[k]->status)
295  {
296  case VStatus::UnDiscovered:
297  this->m_E[vindex][k]->status = EStatus::Tree;
298  this->m_V[k]->parent = vindex;
299  this->DFS(k, clock);
300  break;
301  case VStatus::Discovered:
302  this->m_E[vindex][k]->status = EStatus::Backward;
303  break;
304  case VStatus::Visited:
305  this->m_E[vindex][k]->status = (this->m_V[vindex]->d_time < this->m_V[k]->d_time) ?
306  EStatus::Forward : EStatus::Cross;
307  break;
308  }
309  }
310 
311  this->m_V[vindex]->status = VStatus::Visited;
312  this->m_V[vindex]->f_time = ++clock;
313 }
